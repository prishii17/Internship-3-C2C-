Q: function overloading in child class


#include <iostream>
using namespace std;

class Account
{
protected:
    int acc_no, code;
    char acc_type[20];
    char name[30];
    float balance;

public:
    void getData()
    {
        cout << "Enter Account Number: ";
        cin >> acc_no;

        cout << "Enter Account Type: ";
        cin >> acc_type;

        cout << "Enter Name: ";
        cin >> name;

        cout << "Enter Account Code: ";
        cin >> code;

        cout << "Enter Balance: ";
        cin >> balance;
    }

    void display()
    {
        cout << "\nAccount Number : " << acc_no;
        cout << "\nAccount Type   : " << acc_type;
        cout << "\nName           : " << name;
        cout << "\nAccount Code   : " << code;
        cout << "\nBalance        : " << balance;
    }
};

class Transaction : public Account
{
public:
    void payment()
    {
        cout << "\nGeneral Payment Processed.";
    }

    void payment(float amount)
    {
        balance = balance - amount;
        cout << "\nPayment of Rs. " << amount << " Processed.";
    }

    void payment(float amount, int cashback)
    {
        balance = balance - amount + cashback;
        cout << "\nPayment of Rs. " << amount
             << " Processed with Cashback Rs. "
             << cashback;
    }
};

int main()
{
    Transaction t;

    t.getData();

    t.payment();
    t.display();

    t.payment(1000);
    t.display();

    t.payment(2000, 100);
    t.display();

    return 0;
}
