operator overloading.2
for multiplication

CODE:

#include <iostream>
using namespace std;

class abc {
public:
    int num;
    
    // Fixed operator+ to return an abc object
    abc operator+ (abc obj) {
        abc temp;
        temp.num = num + obj.num;
        return temp;
    }
};

int main() {
    abc s1, s2, s3;
    
    s1.num = 10;
    s2.num = 20;
    
    // Fixed: s1 + s2 returns an 'abc' object, which matches 's3'
    s3 = s1 + s2;
    
    cout << "number is: " << s3.num;
    
    return 0;
}

OUTPUT:

number is: 30
